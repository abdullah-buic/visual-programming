using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp4
{
    class Program
    {
        static void Main(string[] args)
        {
            
            int[] array = new int[10];
            Console.WriteLine("Please Enter any 10 Numbers is the Array");
            for (int i=0; i<10; i++)
            {
                array[i]= Convert.ToInt32(Console.ReadLine());
            }

            int max = 0;
            int min = 15000;
            int secondMax = 0;
            int secondMin = 0;
            for (int i = 0; i < 10; i++)
             {
                 if (array[i]>max)
                 {
                    secondMax = max;
                     max = array[i];
                 }
             }

            for (int j = 0; j < 10; j++)
            {
                if (min > array[j])
                {
                    secondMin = min;
                    min = array[j];
                }
            }

            Console.WriteLine("\n\nDisplaying the Second Smallest Value");
            System.Console.WriteLine(min);

            Console.WriteLine("Displaying the Second Largest Value");
            System.Console.WriteLine(secondMax);

            Console.Read();
        }
    }
}
